    <script src="https://cdn.jsdelivr.net/npm/retell-client-js-sdk@latest/dist/index.umd.js"></script>

    <script>

        const retellWebClient = new RetellWebClient();

        const startBtn = document.getElementById('start-btn');

        const stopBtn = document.getElementById('stop-btn');

        const statusText = document.getElementById('status-text');

        const visualizer = document.getElementById('visualizer');


        // POINT TO YOUR VERCEL API

        const REGISTER_CALL_URL = "/api/register-call"; 


        retellWebClient.on("call_started", () => {

            statusText.innerText = ">> RECORDING ACTIVE";

            statusText.style.color = "#00ff41";

            startBtn.style.display = 'none';

            stopBtn.style.display = 'inline-block';

            visualizer.style.width = "100%";

        });


        retellWebClient.on("call_ended", () => {

            statusText.innerText = ">> UPLOADING DATA...";

            statusText.style.color = "#666";

            visualizer.style.width = "0%";

            setTimeout(() => {

                location.reload(); // Refresh to reset

            }, 2000);

        });


        async function startAudit() {

            startBtn.innerText = "CONNECTING...";

            startBtn.disabled = true;


            try {

                const response = await fetch(REGISTER_CALL_URL);

                const data = await response.json();


                await retellWebClient.startCall({

                    accessToken: data.access_token,

                });


            } catch (err) {

                console.error(err);

                statusText.innerText = ">> ERROR: CHECK API KEYS";

                statusText.style.color = "#ff3333";

                startBtn.disabled = false;

                startBtn.innerText = "INITIATE SEQUENCE";

            }

        }


        function stopAudit() {

            retellWebClient.stopCall();

        }
